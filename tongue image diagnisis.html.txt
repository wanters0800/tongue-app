        const canvas = document.getElementById('imageCanvas');
        const ctx = canvas.getContext('2d', { willReadFrequently: true });
        const imageUpload = document.getElementById('imageUpload');
        const analyzeBtn = document.getElementById('analyzeBtn');
        const resetBtn = document.getElementById('resetBtn');
        const closeBtn = document.getElementById('closeBtn');
        const statusIndicator = document.getElementById('statusIndicator');
        const instructionOverlay = document.getElementById('instructionOverlay');
        const zoomDisplay = document.getElementById('zoomDisplay');
        const resultsSection = document.getElementById('resultsSection');
        const resultsBody = document.getElementById('resultsBody');
        const exportCsvBtn = document.getElementById('exportCsvBtn');
        const processingModal = document.getElementById('processingModal');

        const maskPreviewBox = document.getElementById('mask-preview');

        let originalImage = null, points = [], isClosed = false, analysisResults = [];
        
        // --- 縮放與平移狀態 ---
        let scale = 1.0;
        let offsetX = 0;
        let offsetY = 0;
        let isDragging = false;
        let lastMouseX = 0;
        let lastMouseY = 0;

        // --- UI 事件綁定 ---
        imageUpload.addEventListener('change', handleImageUpload);
        canvas.addEventListener('mousedown', handleMouseDown);
        window.addEventListener('mousemove', handleMouseMove);
        window.addEventListener('mouseup', handleMouseUp);
        canvas.addEventListener('wheel', handleWheel, { passive: false });
        canvas.addEventListener('contextmenu', e => e.preventDefault());

        resetBtn.addEventListener('click', () => { points = []; isClosed = false; redraw(); });
        closeBtn.addEventListener('click', () => { if (points.length >= 3) { isClosed = true; redraw(); } });
        analyzeBtn.addEventListener('click', runAnalysis);
        exportCsvBtn.addEventListener('click', exportToCsv);

        function handleImageUpload(e) {
            const file = e.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = (ev) => {
                const img = new Image();
                img.onload = () => {
                    originalImage = img;
                    points = []; 
                    isClosed = false;
                    resetZoom();
                    statusIndicator.innerText = "IMAGE LOADED";
                    instructionOverlay.style.display = 'none';
                };
                img.src = ev.target.result;
            };
            reader.readAsDataURL(file);
        }

        function setupCanvas(img) {
            const container = canvas.parentElement;
            const maxW = container.clientWidth - 40;
            const maxH = 550;
            let w = img.width, h = img.height;
            if (w > maxW) { h *= maxW / w; w = maxW; }
            if (h > maxH) { w *= maxH / h; h = maxH; }
            canvas.width = w; canvas.height = h;
        }

        // --- 互動邏輯 ---

        function handleMouseDown(e) {
            if (!originalImage) return;
            if (e.button === 2) {
                isDragging = true;
                lastMouseX = e.clientX;
                lastMouseY = e.clientY;
                canvas.style.cursor = 'grabbing';
                return;
            }
            if (e.button === 0 && !isClosed) {
                const coord = getOriginalCoordinates(e.clientX, e.clientY);
                points.push(coord);
                redraw();
            }
        }

        function handleMouseMove(e) {
            if (isDragging) {
                const dx = e.clientX - lastMouseX;
                const dy = e.clientY - lastMouseY;
                offsetX += dx / scale;
                offsetY += dy / scale;
                lastMouseX = e.clientX;
                lastMouseY = e.clientY;
                redraw();
            }
        }

        function handleMouseUp() {
            isDragging = false;
            canvas.style.cursor = 'crosshair';
        }

        function handleWheel(e) {
            if (!originalImage) return;
            e.preventDefault();
            const delta = e.deltaY > 0 ? 0.9 : 1.1;
            handleZoom(delta, e.clientX, e.clientY);
        }

        function handleZoom(delta, clientX, clientY) {
            if (!originalImage) return;
            const prevScale = scale;
            scale *= delta;
            scale = Math.min(Math.max(scale, 0.5), 10);
            if (clientX !== undefined && clientY !== undefined) {
                const rect = canvas.getBoundingClientRect();
                const mouseX = (clientX - rect.left) / prevScale;
                const mouseY = (clientY - rect.top) / prevScale;
                offsetX -= mouseX * (scale/prevScale - 1) / (scale/prevScale);
                offsetY -= mouseY * (scale/prevScale - 1) / (scale/prevScale);
            }
            zoomDisplay.innerText = `${Math.round(scale * 100)}%`;
            redraw();
        }

        function resetZoom() {
            if (!originalImage) return;
            setupCanvas(originalImage);
            scale = 1.0;
            offsetX = 0;
            offsetY = 0;
            zoomDisplay.innerText = `100%`;
            redraw();
        }

        function getOriginalCoordinates(clientX, clientY) {
            const rect = canvas.getBoundingClientRect();
            let x = (clientX - rect.left) / scale - offsetX;
            let y = (clientY - rect.top) / scale - offsetY;
            return { x, y };
        }

        function redraw() {
            if (!originalImage) return;
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.save();
            ctx.scale(scale, scale);
            ctx.translate(offsetX, offsetY);
            ctx.drawImage(originalImage, 0, 0, canvas.width, canvas.height);
            if (points.length > 0) {
                ctx.beginPath();
                ctx.moveTo(points[0].x, points[0].y);
                points.forEach(p => ctx.lineTo(p.x, p.y));
                if (isClosed) {
                    ctx.closePath();
                    ctx.fillStyle = 'rgba(79, 70, 229, 0.2)';
                    ctx.fill();
                }
                ctx.strokeStyle = '#facc15';
                ctx.lineWidth = 2.5 / scale;
                ctx.stroke();
                ctx.fillStyle = '#ef4444';
                points.forEach(p => {
                    ctx.beginPath();
                    ctx.arc(p.x, p.y, 4.5 / scale, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.strokeStyle = 'white';
                    ctx.lineWidth = 1 / scale;
                    ctx.stroke();
                });
            }
            ctx.restore();
            analyzeBtn.disabled = !isClosed;
        }

        // --- 核心運算法 (本地端處理) ---

        function rgbToHsv(r, g, b) {
            r/=255; g/=255; b/=255; 
            const max = Math.max(r,g,b), min = Math.min(r,g,b), d = max-min;
            let h, s, v = max; s = max === 0 ? 0 : d / max;
            if (max === min) h = 0;
            else {
                if(max===r) h=(g-b)/d+(g<b?6:0); else if(max===g) h=(b-r)/d+2; else h=(r-g)/d+4;
                h/=6;
            }
            return { h, s, v };
        }

        function getIQRRange(arr) {
            if (arr.length === 0) return { min: 0, max: 1 };
            const sorted = [...arr].sort((a, b) => a - b);
            const q1 = sorted[Math.floor(sorted.length * 0.25)];
            const q3 = sorted[Math.floor(sorted.length * 0.75)];
            const iqr = q3 - q1;
            return { min: q1 - 1.5 * iqr, max: q3 + 1.5 * iqr };
        }

        function calculateCircularMean(hues) {
            if (hues.length === 0) return 0;
            let sumSin = 0, sumCos = 0;
            for (let h of hues) {
                const rad = h * 2 * Math.PI;
                sumSin += Math.sin(rad); sumCos += Math.cos(rad);
            }
            let avgH = Math.atan2(sumSin / hues.length, sumCos / hues.length) / (2 * Math.PI);
            if (avgH < 0) avgH += 1;
            return avgH;
        }

        function calculateStats(arr) {
            if (arr.length === 0) return { mean: 0, std: 0 };
            const n = arr.length;
            const mean = arr.reduce((a, b) => a + b, 0) / n;
            const std = Math.sqrt(arr.reduce((a, b) => a + Math.pow(b - mean, 2), 0) / n);
            return { mean, std };
        }

        function createMask(w, h, pts) {
            const temp = document.createElement('canvas');
            temp.width = w; temp.height = h;
            const tCtx = temp.getContext('2d');
            tCtx.beginPath(); tCtx.moveTo(pts[0].x, pts[0].y);
            pts.forEach(p => tCtx.lineTo(p.x, p.y));
            tCtx.closePath(); tCtx.fillStyle = 'white'; tCtx.fill();
            return tCtx.getImageData(0, 0, w, h).data;
        }

        function calculateFeatures(imgData, mask) {
            const data = imgData.data, w = imgData.width, h = imgData.height;
            let roiPixelsCount = 0;
            const grid = new Int8Array(w * h).fill(-1);

            for (let i = 0; i < data.length; i += 4) {
                const idx = i / 4;
                if (mask[i] > 128) {
                    roiPixelsCount++;
                    const r = data[i], g = data[i+1], b = data[i+2];
                    const hsv = rgbToHsv(r, g, b);
                    if (hsv.v > 0.92 || (hsv.h >= 0.52 && hsv.h <= 0.78)) {
                        grid[idx] = 5; continue;
                    }
                    const isYellowHue = (hsv.h >= 0.05 && hsv.h <= 0.21);
                    if (isYellowHue && hsv.s > 0.48 && hsv.v > 0.50) grid[idx] = 2;
                    else if (isYellowHue && hsv.s >= 0.30 && hsv.s <= 0.48 && hsv.v > 0.45) grid[idx] = 6;
                    else if (isYellowHue && hsv.s >= 0.22 && hsv.s < 0.30 && hsv.v > 0.50) grid[idx] = 3;
                    else if (hsv.s < 0.28 && hsv.v > 0.62) grid[idx] = 1;
                    else if (hsv.v < 0.35) grid[idx] = 4;
                    else grid[idx] = 0;
                }
            }

            let finalWhite = 0, finalDeepYellow = 0, finalModerateYellow = 0, finalUncertain = 0, finalCrack = 0;
            const bodyPixels = [];
            for (let i = 0; i < w * h; i++) {
                if (grid[i] === -1 || grid[i] === 5) continue;
                const type = grid[i];
                if (type === 0) { 
                    const bI = i * 4;
                    bodyPixels.push({ r: data[bI], g: data[bI+1], b: data[bI+2], hsv: rgbToHsv(data[bI], data[bI+1], data[bI+2]) });
                    continue;
                }
                const x = i % w, y = Math.floor(i / w);
                let ns = 0;
                for (let dy = -1; dy <= 1; dy++) {
                    for (let dx = -1; dx <= 1; dx++) {
                        if (dx === 0 && dy === 0) continue;
                        const nx = x + dx, ny = y + dy;
                        if (nx >= 0 && nx < w && ny >= 0 && ny < h && grid[ny * w + nx] === type) ns++;
                    }
                }
                if (ns >= 2) {
                    if (type === 1) finalWhite++; else if (type === 2) finalDeepYellow++; else if (type === 6) finalModerateYellow++; else if (type === 3) finalUncertain++; else if (type === 4) finalCrack++;
                } else {
                    const bI = i * 4;
                    bodyPixels.push({ r: data[bI], g: data[bI+1], b: data[bI+2], hsv: rgbToHsv(data[bI], data[bI+1], data[bI+2]) });
                }
            }

            if (bodyPixels.length < 20) return null;

            const hR = getIQRRange(bodyPixels.map(p => p.hsv.h));
            const sR = getIQRRange(bodyPixels.map(p => p.hsv.s));
            const vR = getIQRRange(bodyPixels.map(p => p.hsv.v));
            const finalBody = bodyPixels.filter(p => p.hsv.h >= hR.min && p.hsv.h <= hR.max && p.hsv.s >= sR.min && p.hsv.s <= sR.max && p.hsv.v >= vR.min && p.hsv.v <= vR.max);
            
            const meanH = calculateCircularMean(finalBody.map(p => p.hsv.h));
            const statS = calculateStats(finalBody.map(p => p.hsv.s));
            const statV = calculateStats(finalBody.map(p => p.hsv.v));
            const statR = calculateStats(finalBody.map(p => p.r));
            const statG = calculateStats(finalBody.map(p => p.g));
            const statB = calculateStats(finalBody.map(p => p.b));

            const yellowCoatingSum = finalModerateYellow + finalDeepYellow;
            const totalCoatingVal = ((finalWhite + yellowCoatingSum) / roiPixelsCount * 100);
            let pattern = "少苔 (Scanty)";
            if (totalCoatingVal > 35) pattern = "厚苔 (Thick)"; else if (totalCoatingVal >= 15) pattern = "薄苔 (Thin)";

            return {
                id: Date.now(), patientId: document.getElementById('patientId').value, imageId: document.getElementById('imageId').value,
                rgb_mean: [statR.mean, statG.mean, statB.mean], rgb_std: [statR.std, statG.std, statB.std],
                hsv_val: [meanH * 360, statS.mean * 100, statV.mean * 100], hsv_std: [0, statS.std * 100, statV.std * 100],
                total_coating: totalCoatingVal.toFixed(1) + "%", white_coating: (finalWhite / roiPixelsCount * 100).toFixed(1) + "%",
                yellow_coating: (yellowCoatingSum / roiPixelsCount * 100).toFixed(1) + "%", 
                uncertain: (finalUncertain / roiPixelsCount * 100).toFixed(1) + "%",
                crack_area: (finalCrack / roiPixelsCount * 100).toFixed(1) + "%", 
                pattern, timestamp: new Date().toLocaleString(), vals: finalBody.map(p => p.hsv.v)
            };
        }

        function runAnalysis() {
            if (!isClosed || !originalImage) return;
            processingModal.classList.remove('hidden');
            setTimeout(() => {
                const imgData = ctx.getImageData(0, 0, canvas.width, canvas.height);
                const maskData = createMask(canvas.width, canvas.height, points);
                const stats = calculateFeatures(imgData, maskData);
                if (stats) {
                    updatePreviews(maskData, stats.vals, canvas.width, canvas.height);
                    addResultToTable(stats);
                    resultsSection.classList.remove('hidden');
                    resultsSection.scrollIntoView({ behavior: 'smooth' });
                } else { alert("分析失敗，有效數據不足。"); }
                processingModal.classList.add('hidden');
            }, 600);
        }

        function updatePreviews(mask, vals, w, h) {
            maskPreviewBox.innerHTML = '';
            const mCanvas = document.createElement('canvas'); mCanvas.width = w; mCanvas.height = h;
            const mCtx = mCanvas.getContext('2d'); const mData = mCtx.createImageData(w, h);
            for(let i=0; i<mask.length; i+=4) { const v = mask[i]; mData.data[i] = v; mData.data[i+1] = v; mData.data[i+2] = v; mData.data[i+3] = 255; }

            histPreview.innerHTML = '';
            const hCanvas = document.createElement('canvas'); hCanvas.width = 256; hCanvas.height = 100;
            const hCtx = hCanvas.getContext('2d'); const bins = new Array(256).fill(0);
            vals.forEach(v => bins[Math.floor(v * 255)]++);
            const max = Math.max(...bins) || 1; hCtx.strokeStyle = '#4f46e5'; hCtx.lineWidth = 2; hCtx.beginPath();
            bins.forEach((count, i) => { const x = i; const y = 100 - (count / max * 100); if(i===0) hCtx.moveTo(x, y); else hCtx.lineTo(x, y); });
            hCtx.stroke(); histPreview.appendChild(hCanvas);
        }

        function exportToCsv() {
            if (analysisResults.length === 0) return;
            const headers = ["PatientID", "ImageID", "Body_R", "Body_G", "Body_B", "Body_H", "Body_S", "Body_V", "TotalCoating", "White", "LightYellow", "Yellow", "Crack", "Pattern", "Timestamp"];
            let csv = "\uFEFF" + headers.join(",") + "\n";
            analysisResults.forEach(res => {
                csv += [res.patientId, res.imageId, ...res.rgb_mean, ...res.hsv_val, res.total_coating, res.white_coating, res.uncertain, res.yellow_coating, res.crack_area, res.pattern, `"${res.timestamp}"`].join(",") + "\n";
            });
            const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
            const url = window.URL.createObjectURL(blob);
            const a = document.createElement('a'); a.href = url; a.download = `local_tongue_data_${Date.now()}.csv`; a.click();
        }
